`ifndef GEN3
`define GEN3
`include "trans.sv"
`include "generator.sv"

class read_oper extends generator;
  mailbox gen2drv;
  trans tr;
  
  function new(mailbox gen2drv)
  	this.gen2drv = gen2drv;
  endfunction
  
  task run();
    repeat(10) begin
      tr = new();
      if(!tr.randomize with {PRESETn == 1; PWRITE == 0}) begin
        $display("Randomization Failed");
      end else begin
        gen2drv.put(tr);
        $display("[GEN] Generated Read transcation: %0p", tr);
      end
    end
  endtask
endclass
`endif

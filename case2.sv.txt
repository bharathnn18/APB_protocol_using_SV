`ifndef GEN2
`define GEN2
`include "trans.sv"
`include "generator.sv"

class write_oper extends generator;
  mailbox gen2drv;
  trans tr;
  
  function new (mailbox gen2drv);
    this.gen2drv = gen2drv;
  endfunction
  
  task run();
    repeat(10) begin
      tr = new();
      if(!tr.randomize(() with {PRESETn == 1; PWRITE == 1}) begin
        $display("Randomization Failed");
    end else begin
      gen2drv.put(tr);
      $display("[GEN] Generator WRITE transaction: %0p", tr);
    end
  end       
  endtask
`endif
